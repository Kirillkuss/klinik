// Таблица Документ пациента

CREATE TABLE Document(
	id_document serial PRIMARY KEY,          - ИД документа
	type_document VARCHAR ( 40 ) NOT NULL,   - Тип документа
	seria VARCHAR ( 20 ) NOT NULL ,			 - Серия документа
	numar VARCHAR ( 20 ) NOT NULL UNIQUE,    - Номер документа
	snils VARCHAR ( 20 ) UNIQUE,			 - СНИЛС документа 
	polis VARCHAR ( 20 ) UNIQUE				 - Полис документа
);

//Таблица Пациент

CREATE TABLE Patient(
	id_patient serial PRIMARY KEY,			- ИД пациента
	surname VARCHAR ( 30 ) NOT NULL,		- Фамилия Пациента
	name VARCHAR ( 30 ) NOT NULL,			- Имя
	full_name VARCHAR ( 30 ) NOT NULL,		- Отчество
	gender bool NOT NULL,					- Пол
	phone  VARCHAR ( 12 ) NOT NULL UNIQUE,	- Телефон
	address VARCHAR ( 100 ) NOT NULL ,		- Адресс проживания
	document_id int unique references public.Document ( id_document )	
);

//  Таблица Доктор

CREATE TABLE Doctor(
	id_doctor serial PRIMARY KEY,		- ИД доктора
	surname VARCHAR ( 60 ) NOT NULL,	- Фамилия
	name VARCHAR ( 60 ) NOT NULL,		- Имя
	full_name VARCHAR ( 60 ) NOT NULL	- Отчество
);

//Таблица Жалобы

CREATE TABLE Complaint(
	id_complaint serial PRIMARY KEY,						- ИД жалобы
	functional_impairment VARCHAR ( 100 ) NOT NULL UNIQUE   - Наименование жалобы
);

//Таблица реабилитационное лечение 

CREATE TABLE Rehabilitation_solution(
	id_rehabilitation_solution serial PRIMARY KEY,	- ИД реабилитационного лечения
	name VARCHAR ( 100 ) NOT NULL UNIQUE,			- Наименование ребилитационного лечения	
	survey_plan VARCHAR ( 255 ) NOT NULL			- План лечения
);

// Таблица Карта пациента

CREATE TABLE Card_patient(
	id_card_patient serial PRIMARY KEY,	 							- ИД карты Пациента
	diagnosis VARCHAR ( 50 ) NOT NULL,	 							- Диагнос пациента
	allergy   bool NOT NULL,			 							- Аллергия да/нет
	note VARCHAR ( 255 ) NOT NULL, 		 							- Примечания
	сonclusion VARCHAR ( 255 ) NOT NULL, 							- Заключение  
	complaint_id int8,					 						    - ИД Жалобы
	pacient_id int unique references public.Patient ( id_patient ), - ИД пациента
	FOREIGN KEY (complaint_id) REFERENCES public.Complaint(id_complaint) ON DELETE CASCADE ON UPDATE CASCADE	
);

// Таблица Медикаментозное лечение (ПИТРС)

CREATE TABLE Drug_treatment(
    id_drug serial PRIMARY KEY,   		 - ИД Медикаментозного лечения
    name VARCHAR ( 255 ) NOT NULL UNIQUE - Наименование медикаментозного ленения
);

//Таблица Лечение 

CREATE TABLE Treatment (
	id_treatment serial PRIMARY KEY,				- ИД Лечения
	time_start_treatment timestamp(6) NOT NULL,		- Дата и время начала лечения
	end_time_treatment timestamp(6) NOT NULL,		- Дата и время окончания лечения
	drug_id int8 NOT NULL, 							- ИД медикаментозного лечения
	dosage_drug_treatment VARCHAR ( 255 ) NOT NULL,	- Дозировка медикаментозного лечения 
	card_patient_id int8 NOT NULL,					- ИД карты пациента 
	doctor_id int8 NOT NULL,						- ИД доктора
	rehabilitation_solution_id int8 NOT NULL,		- ИД реабилитационного лечения 
	FOREIGN KEY (rehabilitation_solution_id) REFERENCES public.Rehabilitation_solution(id_rehabilitation_solution),
	FOREIGN KEY (doctor_id) REFERENCES public.Doctor(id_doctor),	
    FOREIGN KEY (card_patient_id) REFERENCES public.Card_patient(id_card_patient),
	FOREIGN KEY (drug_id) REFERENCES public.Drug_treatment(id_drug)	
);

//Таблица Запись пациента к врачу

CREATE TABLE Record_patient(
	id_record serial PRIMARY KEY,			- ИД записи пациента к врачу
	date_record timestamp(6) NOT NULL,		- Дата и время записии 
	date_appointment timestamp(6) NOT NULL,	- Дата и время приема
	number_room int4 NOT NULL,				- Номер кабинета
	doctor_id int8 NOT NULL,				- Ид доктора
	card_patient_id  int8 NOT NULL,			- ИД карты пациента
	FOREIGN KEY (doctor_id) REFERENCES public.Doctor(id_doctor) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (card_patient_id) REFERENCES public.Card_patient(id_card_patient) ON DELETE CASCADE ON UPDATE CASCADE
);

// Таблица поджалобы
CREATE TABLE Type_complaint(
	id_type_complaint serial PRIMARY KEY,
	name VARCHAR ( 150 ) NOT NULL UNIQUE,
	complaint_id int8 NOT NULL,
	FOREIGN KEY (complaint_id) REFERENCES public.Complaint(id_complaint)
);

//Таблица лекарства
CREATE TABLE Grug(
	id_dr serial PRIMARY KEY,
	name VARCHAR ( 255 ) NOT NULL UNIQUE,
	drug_id int8 NOT NULL,
	FOREIGN KEY (drug_id) REFERENCES public.Drug_treatment(id_drug)
);
